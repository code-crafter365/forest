keywords = {Naringin, Bismuth nanoparticles, Anticancer},
abstract = {Bismuth-naringin (BiNar) nanoparticles were developed using a straightforward sunlight-assisted photochemical method to enhance the properties of naringin. The resulting BiNar colloid consists in semi-spherical particles of approximately 30 nm in size with a negative surface charge. Complexation was elucidated using various spectrometric techniques. The anticancer properties of both naringin and the nanoparticles were evaluated by performing MTT colorimetric assays in human cervical cancer cells. BiNar significantly reduced the viability of cancer cells, underscoring its therapeutic potential.}
