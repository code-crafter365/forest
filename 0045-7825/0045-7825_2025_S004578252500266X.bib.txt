keywords = {Deep-Material Network, Stochastic, Composites, Stochastic Volume Elements (SVEs), Homogenisation},
abstract = {The Interaction-Based Deep Material Network (IB-DMN) is reformulated to decouple the phase volume fraction from the topological parameters of the IB-DMN. Since the phase volume fraction is no longer influenced by the topological parameters, on the one hand the stochastic IB-DMN can predict the response of arbitrary phase volume fraction, and on the other hand the stochastic IB-DMN can be constructed by introducing uncertainties to the topological parameters of a reference IB-DMN, which is trained using data obtained from full-field linear elastic homogenisation, allowing to capture the variability resulting from the micro-structure organisation such as a phase clustering. The non-linear predictions of the proposed stochastic IB-DMN are compared to those from Direct Numerical Simulation (DNS) on 2D Stochastic Volume Elements (SVEs) of unidirectional fibre-reinforced matrix composites in a finite-strain setting. The results from in-plane uni-axial stress and shear tests show that the proposed stochastic IB-DMN is capable of reproducing random non-linear responses with the same stochastic characteristics as the predictions of the DNS conducted on SVE realisations.}
